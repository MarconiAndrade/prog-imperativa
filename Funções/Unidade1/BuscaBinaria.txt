public static int buscaBinaria(int[] v, int tam, int x){
    int inicio = 0, fim = tam - 1, meio;
    while(inicio <= fim){
        meio = (inicio + fim)/2;
        if(x == v[meio]){
            return meio;
        }
        else if(x < v[meio]){
            fim = meio - 1;
        }
        else{
            inicio = meio + 1;
        }
    }
    return -1;
}

public static void selectionSort(int[] v, int tam){
    int posMenor, aux;
    for(int i = 0; i <= tam-2; i++){
        posMenor = i;
        for(int j = i+1; j <= tam-1; j++){
            if(v[j] < v[posMenor]){
                posMenor = j;
            }
        }
        if(i != posMenor){
            aux = v[i];
            v[i] = v[posMenor];
            v[posMenor] = aux;
        }
    }
}

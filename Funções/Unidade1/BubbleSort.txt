public static void bubbleSortPorID(Produto[] v, int tam){
    Produto aux;
    boolean trocou = true;
    int fim = tam - 2, posUltimaTroca = tam - 2;
    while(trocou){
        trocou = false;
        for(int i = 0; i <= fim; i++){
            if(v[i].id > v[i + 1].id){
                trocou = true;
                aux = v[i];
                v[i] = v[i + 1];
                v[i + 1] = aux;
                posUltimaTroca = i;
            }
        }
        fim = posUltimaTroca - 1;
    }
}

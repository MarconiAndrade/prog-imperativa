public static int removerTodos(int[] v, int tam, int x){
    int desl = 0;
    for(int i = 0; i < tam; i++){
        if(v[i] == x){
            desl += 1;
        }
        else{
            v[i - desl] = v[i];
        }
    }
    return tam - desl;
}

public static int removerTodos(int[] v, int tam, int x){
    int novoTam = 0;
    for(int i = 0; i < tam; i++){
        if(v[i] != x){
            v[novoTam] = v[i];
            novoTam++;
        }
    }
    return novoTam;
}
